# C++ login

조건문 및 반복문을 이용하여 로그인 프로그램 작성해보도록 하겠다.
​
```cpp
#include <iostream>
#include <string>

using namespace std;

int main() {

	//간단한 로그인 로직 프로그램
	//데이터베이스에 저장된 데이터라 가정.
	string db_user = "admin";
	string db_password = "1234";

	//입력받을 유저 아이디.
	string input_user = "";
	//입력받을 유저 비밀번호
	string input_password = "";
	//로그인 실패 체크
	int count = 0;
	const int failCount = 5;
	bool islogined = false;

	cout << "간단한 로그인 프로그램 입니다." << endl;

	while (!islogined) {
		if (count > failCount) {
			cout << "5회 초과하여 계정이 잠겼습니다. " << endl;
			islogined = true;
			continue;
		}
		else if(count <= failCount && count > 0) {
			cout << "실패한 로그인 횟수 :" << count << " 회" << endl;
			input_user = "";
			input_password = "";
		}
		cout << "아이디를 입력해주세요 :";
		cin >> input_user;
		cout << "비밀번호를 입력해주세요 :";
		cin >> input_password;
		//아이디가 일치할 경우.
		if (db_user == input_user) {
			//로그인 성공한 경우.
			if (db_password == input_password) {
				islogined = true;
			}
			//비밀번호가 틀린경우.
			else {
				cout << "비밀번호가 틀렸습니다." << endl;
			}
		}
		//아이디가 일치하지 않을경우.
		else {
			cout << "존재하지 않는 아이디입니다." << endl;
		}
		count++;
		cout << "" << endl;
	}
	count > failCount ? cout : cout << "로그인에 성공하였습니다. 축하합니다." << endl;
	return 0;
}
```
